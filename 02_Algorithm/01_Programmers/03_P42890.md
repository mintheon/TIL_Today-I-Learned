    private boolean check(String[][] relation, int rowsize, int colsize, int subset) {
        for ( int i = 0; i < rowsize -1 ; ++i) {
            for (int j = i+1; j < rowsize; ++j){
                boolean isSame = true;
                for (int k = 0 ; k < colsize; ++k){
                    if((subset & 1<<k) == 0){
                        continue;
                    }
                    if(relation[i][k].equals(relation[j][k]) == false){
                        isSame = false;
                        break;
                    }
                }
                if(isSame)
                    return false;
            }
        }
        return true;
    }
